/*
 * UID Reader — Smart Night Market
 * Tap any NFC card → prints its UID to Serial monitor
 *
 * Use this to find your card's UID before provisioning.
 * Steps:
 *   1. Rename this file to main.cpp  (back up real main.cpp as main.cpp.bak first)
 *   2. Upload to ESP32
 *   3. Open Serial monitor at 115200 baud
 *   4. Tap your card — UID appears in Serial
 *   5. Copy the UID
 *   6. Rename this file back to read_uid.cpp.txt
 *   7. Restore main.cpp.bak → main.cpp
 */

#include <Arduino.h>
#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN   21
#define RST_PIN  22
#define MOSI_PIN 23
#define MISO_PIN 19
#define SCK_PIN  18

MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
    Serial.begin(115200);
    delay(500);
    Serial.println("\n--- UID Reader Ready ---");
    Serial.println("Tap a card to see its UID...\n");

    SPI.begin(SCK_PIN, MISO_PIN, MOSI_PIN, SS_PIN);
    mfrc522.PCD_Init();
}

void loop() {
    if (!mfrc522.PICC_IsNewCardPresent() || !mfrc522.PICC_ReadCardSerial()) {
        return;
    }

    String uid = "";
    for (byte i = 0; i < mfrc522.uid.size; i++) {
        if (mfrc522.uid.uidByte[i] < 0x10) uid += "0";
        uid += String(mfrc522.uid.uidByte[i], HEX);
    }
    uid.toUpperCase();

    Serial.println("Card UID: " + uid);
    Serial.println("Copy this UID into provision.cpp.txt");
    Serial.println("-----------------------------------");

    mfrc522.PICC_HaltA();
    delay(2000);
}
