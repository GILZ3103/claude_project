/*
 * RC522 Diagnostic Sketch
 * Tries reading multiple registers to confirm the chip is responding at all.
 *
 * Usage:
 *   1. Back up main.cpp → main.cpp.bak (if not already)
 *   2. Rename this file to main.cpp
 *   3. Upload to ESP32
 *   4. Open Serial monitor at 115200 baud
 *   5. Paste the output back here
 */

#include <Arduino.h>
#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN   21
#define RST_PIN  22
#define MOSI_PIN 23
#define MISO_PIN 19
#define SCK_PIN  18

MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
    Serial.begin(115200);
    delay(1000);
    Serial.println("\n=== RC522 Diagnostic ===\n");

    Serial.println("Pin assignments:");
    Serial.println("  SS  (CS/SDA) = GPIO 21");
    Serial.println("  SCK          = GPIO 18");
    Serial.println("  MOSI         = GPIO 23");
    Serial.println("  MISO         = GPIO 19");
    Serial.println("  RST          = GPIO 22");
    Serial.println();

    // Hardware reset via RST pin
    Serial.println("Toggling RST pin manually...");
    pinMode(RST_PIN, OUTPUT);
    digitalWrite(RST_PIN, LOW);
    delay(50);
    digitalWrite(RST_PIN, HIGH);
    delay(50);

    Serial.println("Initialising SPI bus...");
    SPI.begin(SCK_PIN, MISO_PIN, MOSI_PIN, SS_PIN);
    SPI.setFrequency(1000000); // 1 MHz — slow, conservative
    delay(50);

    Serial.println("Initialising RC522...");
    mfrc522.PCD_Init();
    delay(100);

    Serial.println("\nReading registers:");
    Serial.println("------------------");

    byte ver = mfrc522.PCD_ReadRegister(MFRC522::VersionReg);
    Serial.print("  VersionReg (0x37) = 0x");
    if (ver < 0x10) Serial.print("0");
    Serial.println(ver, HEX);
    Serial.print("  Interpretation   : ");
    switch (ver) {
        case 0x91: Serial.println("v1.0 — OK"); break;
        case 0x92: Serial.println("v2.0 — OK"); break;
        case 0x88: Serial.println("Fudan FM17522 clone — OK"); break;
        case 0xB2: Serial.println("Counterfeit FM17522 — OK"); break;
        case 0x00: Serial.println("NO RESPONSE — wiring/chip dead"); break;
        case 0xFF: Serial.println("SPI line stuck HIGH — wiring fault"); break;
        default:   Serial.println("Unknown — but chip is responding"); break;
    }

    byte mode = mfrc522.PCD_ReadRegister(MFRC522::ModeReg);
    Serial.print("  ModeReg    (0x11) = 0x");
    if (mode < 0x10) Serial.print("0");
    Serial.println(mode, HEX);

    byte tx   = mfrc522.PCD_ReadRegister(MFRC522::TxControlReg);
    Serial.print("  TxCtrlReg  (0x14) = 0x");
    if (tx < 0x10) Serial.print("0");
    Serial.println(tx, HEX);

    Serial.println("\n=== End of test ===");
}

void loop() {
    delay(10000);
    Serial.println("Re-reading version...");
    byte ver = mfrc522.PCD_ReadRegister(MFRC522::VersionReg);
    Serial.print("  VersionReg = 0x");
    if (ver < 0x10) Serial.print("0");
    Serial.println(ver, HEX);
}
